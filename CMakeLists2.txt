# On macOS, Apple Clang ships without OpenMP,  GCC has built-in OpenMP support ..

if(APPLE)

    # Check if the compiler is Clang (Apple or Homebrew Clang)
    if(CMAKE_CXX_COMPILER_ID MATCHES "Clang|AppleClang")
        find_program(_brew brew)
        if(_brew)
            execute_process(COMMAND ${_brew} --prefix libomp
                            OUTPUT_VARIABLE _libomp_prefix
                            OUTPUT_STRIP_TRAILING_WHITESPACE
                            ERROR_QUIET)
            if(_libomp_prefix AND EXISTS "${_libomp_prefix}")
                set(OpenMP_CXX_FLAGS     "-Xpreprocessor -fopenmp")
                set(OpenMP_C_FLAGS       "-Xpreprocessor -fopenmp")
                set(OpenMP_CXX_LIB_NAMES "omp")
                set(OpenMP_C_LIB_NAMES   "omp")
                set(OpenMP_omp_LIBRARY   "${_libomp_prefix}/lib/libomp.dylib")
                include_directories(SYSTEM "${_libomp_prefix}/include")
                link_directories("${_libomp_prefix}/lib")
                message(STATUS "macOS + Clang: using Homebrew libomp at ${_libomp_prefix}")
            else()
                message(WARNING "macOS + Clang: libomp not found via Homebrew. Run: brew install libomp")
            endif()
        endif()
    else()
        message(STATUS "macOS + ${CMAKE_CXX_COMPILER_ID}: using built-in OpenMP support")
    endif()
endif()
